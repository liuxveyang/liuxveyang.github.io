空结构体大小是多少？

October 2, 2013
C 

/*
    在VC中，空结构体大小是1。
    在gcc中，空结构体大小是0。
*/
#include <cstdio>
struct student 
{
    
}stu;
int main(void)
{
    printf("%d\n", sizeof stu);

    return 0;
}
