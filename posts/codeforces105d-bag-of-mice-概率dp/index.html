using namespace std;
double a[1111][1111];
int main(void)
{
    int w,b;
    #ifndef ONLINE_JUDGE
    freopen("in.txt","r",stdin);
    #endif // ONLINE_JUDGE
    while (~scanf("%d%d",&w,&b))
    {
        for(int i=0;i<=w;++i)for(int j=0;j<=b;++j)a[i][j]=0;
        for(int i=0;i<=b;++i) a[0][i]=0;
        for(int j=1;j<=w;++j) a[j][0]=1;
        for(int i=1;i<=w;++i)
        {
            for(int j=1;j<=b;++j)
            {
                a[i][j]+=(double)i/(i+j);
                if(j>=3)
                    a[i][j]+=a[i][j-3]*(double)j/(i+j)
                    *(double)(j-1)/(i+j-1)
                    *(double)(j-2)/(i+j-2);
                if(i>=2)
                    a[i][j]+=a[i-1][j-2]*(double)j/(i+j)
                    *(double)(j-1)/(i+j-1)
                    *(double)(i)/(i+j-2);
            }
        }
        printf("%.9lf\n",a[w][b]);
    }
    return 0;
}
