       break;
     i = 0;
     getchar();
     while ((c = getchar()) != '\n')
     {
       a[i] = c;
       i++;
     }
     a[i] = '\0';
     for (i = 0; i < strlen(a); i++)
     {
       if (a[i] >= 'A' && a[i] <= 'E')
