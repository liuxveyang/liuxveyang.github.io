编写中断例程7ch：计算word型数据的平方

October 17, 2013
masm

实现计算一个word型数据的平方。

这是安装程序。

assume cs:code

code segment
start:
    mov ax, cs
    mov ds, ax
    mov si, offset sqr

    mov ax, 0
    mov es, ax
    mov di, 200h
    mov cx, offset sqrend - offset sqr
    cld
    rep movsb

    mov ax, 0
    mov es, ax
    mov word ptr es:[7ch * 4+0], 200h
    mov word ptr es:[7ch * 4 + 2], 0

    mov ax, 4c00h
    int 21h

sqr:
    mul ax
    iret
sqrend:
    nop


code ends

end start

这是应用程序。

assume cs:code

code segment
start:
    mov ax, 2
    int 7ch
    add ax, ax
    adc dx, dx

    mov ax, 4c00h
    int 21h
code ends

end start
