using namespace std;
int main(void)
{
    int n;
    while (~scanf("%d",&n))
    for (int i=1;i<=n*n/2;++i)
    {
        for (int j=1;j<=n/2;++j)
        {
            printf("%d %d ",i,n*n-i+1);
            ++i;
        }
        i--;
        printf("\n");
    }
    return 0;
}

o(╯□╰)o 官方解题报告上面写的很明白 1　　n^2 2　　n^2-1 ..　　.. n^2/2　　n^2⁄2+1 然后这就是n^2/2组，每一组的和都相同。n^2/2/(n/2) = n组，所以我们只需要从这n^2/2组里面选择n/2组就满足了一个解，这样，恰好可以选出n组来。
