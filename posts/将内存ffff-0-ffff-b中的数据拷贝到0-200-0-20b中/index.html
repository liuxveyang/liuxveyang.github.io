将内存ffff:0~ffff:b中的数据拷贝到0:200~0:20b中

October 4, 2013
masm

我是按照字，也就是2个字节拷贝的。

这样就可以让循环减半==

assume cs:sad
sad segment
start:
    mov ax, 0ffffh
    mov ds, ax
    mov bx, 0h
    mov ax, 20h
    mov es, ax
    mov cx, 7h
s:
    mov ax, ds:[bx]
    mov es:[bx], ax
    add bx, 2
    loop s
 
    mov ax, 4c00h
    int 21h
sad ends
end
