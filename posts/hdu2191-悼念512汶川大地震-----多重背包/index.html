using namespace std;
int n, V, t, f[111], c[111], w[111], m[111];
void zeropack(int c, int w)
{
    for (int v=V; v>=c; --v)
    {
        f[v]=max(f[v],f[v-c]+w);
    }
    return;
}
void completepack(int c, int w)
{
    for (int v=c;v<=V;++v)
        f[v]=max(f[v],f[v-c]+w);
    return;
}
void multipack(int c, int w, int m)
{
    if (c*m >= V)
    {
        completepack(c,w); return;
    }
    int k=1;
    while (k<m)
    {
        zeropack(c*k,w*k); m-=k; k*=2;
    }
    zeropack(c*m,w*m);
    return;
}

        for (int i=0;i<n;++i)
        {
            multipack(c[i],w[i],m[i]);
        }
