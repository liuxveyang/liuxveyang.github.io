typedef long long int LL;
const int MAXN =  0x3f3f3f3f;
const int  MIN =  -0x3f3f3f3f;

int m;
LL power(LL a, LL k){
  LL ans = 1;
  while (k){
    if (k&1){
      ans = (ans * a) % m; k--;
    }
    k >>= 1; a = (a*a)%m;
  }
  return ans;
}
