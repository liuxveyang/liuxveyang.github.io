using namespace std;
char a[111111], b[11111];
//#define LL long long
#define LL __int64
int main(void)
{
    #ifndef ONLINE_JUDGE
    freopen("in.txt", "r", stdin );
    #endif // ONLINE_JUDGE
    while (~scanf("%s", a))
    {
        int n; scanf("%d", &n);
        int l, r, pl, pr, tmp = 0;
        LL ans = 0, bns = 0;
        scanf("%d%d", &pl, &pr);
        bns += (pr - pl + 1);
        ans += (pr - pl + 1 + 2);
        for (int so = 0; so < n-1; ++so)
        {
            scanf("%d%d", &l, &r);
            bns += (r - l + 1);
            tmp = 0; int pos = pl, i;
            for (i = l; i < r && pos < pr; ++i)
            {
                if (a[i] == a[pos]) tmp++, pos++;
                else break;
            }
            sprintf(b, "%d", tmp);
            ans += strlen(b);
            ans += (r-l-(i-l)); ans += 2;
            pl =l, pr = r;
        }
//        printf("%lld %lld\n", bns, ans);
//        printf("bns = %lld\n",bns);
//        printf("ans = %lld\n",ans);
//        printf("%lld %lld\n", bns, ans);
        cout << bns << " " << ans << endl;
    }
    return 0;
}
